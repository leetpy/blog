# 题目

[https://leetcode.cn/problems/count-covered-buildings/description/?envType=daily-question&envId=2025-12-11]()

# 思路

记录每个点 x, y 出现的最小最大值，在这个之间的即满足要求

```go
func countCoveredBuildings(n int, buildings [][]int) int {
	xMap := make(map[int][]int)
	yMap := make(map[int][]int)
	for _, v := range buildings {
		x, y := v[0], v[1]
		if _, ok := xMap[x]; !ok {
			xMap[x] = []int{y, y}
		} else {
			xMap[x][0] = min(y, xMap[x][0])
			xMap[x][1] = max(y, xMap[x][1])
		}

		if _, ok := yMap[y]; !ok {
			yMap[y] = []int{x, x}
		} else {
			yMap[y][0] = min(x, yMap[y][0])
			yMap[y][1] = max(x, yMap[y][1])
		}
	}

	ans := 0
	for _, v := range buildings {
		x, y := v[0], v[1]
		if xMap[x][0] < y && y < xMap[x][1] && yMap[y][0] < x && x < yMap[y][1] {
			ans++
		}
	}

	return ans
}
```
